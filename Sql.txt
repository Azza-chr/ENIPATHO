-- ===== ÉTAPE 1: SUPPRIMER TOUT =====
DROP TABLE IF EXISTS etudiant_enseignants;

DROP TABLE IF EXISTS academic_choix_reponses;

DROP TABLE IF EXISTS academic_quizzes;

-- ===== ÉTAPE 2: CRÉER TABLES UTILISATEURS =====

CREATE TABLE utilisateur (
    usr_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(100) NOT NULL,
    prenom VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    mdp VARCHAR(255) NOT NULL,
    actif BOOLEAN DEFAULT TRUE,
    photo LONGBLOB,
    dtype VARCHAR(50),
    theme VARCHAR(20) DEFAULT 'light',
    langue VARCHAR(10) DEFAULT 'fr',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE admin (
    id BIGINT PRIMARY KEY,
    CONSTRAINT fk_admin_user FOREIGN KEY (id) REFERENCES utilisateur(usr_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE chef_departement (
    id BIGINT PRIMARY KEY,
    departement VARCHAR(100),
    CONSTRAINT fk_chef_user FOREIGN KEY (id) REFERENCES utilisateur(usr_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE matiere (
    matiere_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(150) NOT NULL,
    code VARCHAR(50) UNIQUE,
    description TEXT,
    credits INT DEFAULT 0
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE groupe (
    groupe_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    code CHAR(1) NOT NULL,
    niveau INT NOT NULL,
    max_etudiants INT DEFAULT 30,
    nombre_etudiants INT DEFAULT 0,
    CONSTRAINT chk_code CHECK(code IN ('A', 'B', 'C', 'D')),
    CONSTRAINT chk_niveau CHECK(niveau IN (1, 2, 3)),
    UNIQUE KEY uk_groupe (code, niveau)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE enseignant (
    id BIGINT PRIMARY KEY,
    departement VARCHAR(100),
    specialite VARCHAR(100),
    matiere_id BIGINT,
    max_groupes INT DEFAULT 3,
    bio TEXT,
    phone VARCHAR(20),
    office VARCHAR(50),
    CONSTRAINT fk_enseignant_user FOREIGN KEY (id) REFERENCES utilisateur(usr_id) ON DELETE CASCADE,
    CONSTRAINT fk_enseignant_matiere FOREIGN KEY (matiere_id) REFERENCES matiere(matiere_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE etudiant (
    id BIGINT PRIMARY KEY,
    niveau INT NOT NULL,
    groupe CHAR(1) NOT NULL,
    total_badges INT DEFAULT 0,
    score INT DEFAULT 0,
    progression_percentage INT DEFAULT 0,
    total_heures_etude INT DEFAULT 0,
    CONSTRAINT fk_etudiant_user FOREIGN KEY (id) REFERENCES utilisateur(usr_id) ON DELETE CASCADE,
    CONSTRAINT chk_etud_niveau CHECK(niveau IN (1, 2, 3)),
    CONSTRAINT chk_etud_groupe CHECK(groupe IN ('A', 'B', 'C', 'D'))
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE enseignant_groupes (
    enseignant_id BIGINT NOT NULL,
    groupe_id BIGINT NOT NULL,
    matiere_id BIGINT,
    PRIMARY KEY (enseignant_id, groupe_id),
    CONSTRAINT fk_eg_enseignant FOREIGN KEY (enseignant_id) REFERENCES enseignant(id) ON DELETE CASCADE,
    CONSTRAINT fk_eg_groupe FOREIGN KEY (groupe_id) REFERENCES groupe(groupe_id) ON DELETE CASCADE,
    CONSTRAINT fk_eg_matiere FOREIGN KEY (matiere_id) REFERENCES matiere(matiere_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE etudiant_enseignants (
    etudiant_id BIGINT NOT NULL,
    enseignant_id BIGINT NOT NULL,
    PRIMARY KEY (etudiant_id, enseignant_id),
    CONSTRAINT fk_ee_etudiant FOREIGN KEY (etudiant_id) REFERENCES etudiant(id) ON DELETE CASCADE,
    CONSTRAINT fk_ee_enseignant FOREIGN KEY (enseignant_id) REFERENCES enseignant(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ===== ÉTAPE 3: CRÉER TABLES ACADÉMIQUES =====

CREATE TABLE academic_semestres (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(10) NOT NULL UNIQUE,
    libelle VARCHAR(100) NOT NULL,
    ordre INT NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_matieres (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(50) UNIQUE NOT NULL,
    nom VARCHAR(150) NOT NULL,
    description TEXT,
    enseignant_nom VARCHAR(100),
    semestre_id BIGINT NOT NULL,
    CONSTRAINT fk_matiere_semestre FOREIGN KEY (semestre_id) REFERENCES academic_semestres(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_chapitres (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    titre VARCHAR(200) NOT NULL,
    description TEXT,
    ordre INT NOT NULL,
    published BOOLEAN DEFAULT TRUE,
    matiere_id BIGINT NOT NULL,
    CONSTRAINT fk_chapitre_matiere FOREIGN KEY (matiere_id) REFERENCES academic_matieres(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_ressources (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    titre VARCHAR(200) NOT NULL,
    description TEXT,
    file_url VARCHAR(500),
    mime_type VARCHAR(50),
    type VARCHAR(50),
    chapitre_id BIGINT NOT NULL,
    CONSTRAINT fk_ressource_chapitre FOREIGN KEY (chapitre_id) REFERENCES academic_chapitres(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_quizzes (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    titre VARCHAR(200) NOT NULL,
    description TEXT,
    duration_minutes INT DEFAULT 20,
    passing_score INT DEFAULT 60,
    chapitre_id BIGINT,
    cours_id BIGINT,
    CONSTRAINT fk_quiz_chapitre FOREIGN KEY (chapitre_id) REFERENCES academic_chapitres(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_questions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    enonce TEXT NOT NULL,
    explication TEXT,
    ordre INT NOT NULL,
    points INT DEFAULT 1,
    quiz_id BIGINT NOT NULL,
    CONSTRAINT fk_question_quiz FOREIGN KEY (quiz_id) REFERENCES academic_quizzes(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE academic_choix_reponses (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    label TEXT NOT NULL,
    correct BOOLEAN DEFAULT FALSE,
    ordre INT NOT NULL,
    question_id BIGINT NOT NULL,
    CONSTRAINT fk_choix_question FOREIGN KEY (question_id) REFERENCES academic_questions(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

SET FOREIGN_KEY_CHECKS=1;
